  265.      * Fonction qui set comme contenu de cellule une somme (par exemple A1:B6;C10)
    266. 

  266.      *

  271.     public function setFormulaSUM($formula) {
    272. 

  272.         $f = (strpos($formula, '=SUM(') === false) ? '=SUM('.$formula.')' : $formula;
    273. 

  273.         // On sauve la formule
    274. 

  274.         $f = str_replace('(', '([.', $f); $f = str_replace(')', '])', $f);
    275. 

  275.         $f = str_replace(';', '];[.', $f); $f = str_replace(':', ':.', $f);
    276. 

  276.         // On en extrait les données
    277. 

  277.         $formula = ltrim($formula, '=SUM(');
    278. 

  278.         $formula = rtrim($formula, ')');
    279. 

  279.         $l = Fonction::getLetters(true);
    280. 

  280.         // On calcule le résultat
    281. 

  281.         $somme = 0;
    282. 

  282.         $blocs = explode(';', $formula);
    283. 

  283.         $cellules = $this->parent->getCells();
    284. 

  284.         foreach ($blocs as $bloc) {
    285. 

  285.             $cells = explode(':', $bloc);
    286. 

  286.             $tab = array();
    287. 

  287.             foreach ($cells as $cell) {
    288. 

  288.                 $cols = ''; $row = '';
    289. 

  289.                 for ($i = 0; $i < strlen($cell); $i++) {
    290. 

  290.                     if (is_int($cell[$i]) || is_float($cell[$i]) || is_numeric($cell[$i])) $row .= $cell[$i];
    291. 

  291.                     else $cols .= $cell[$i];
    292. 

  292.                 }
    293. 

  293.                 // On chope la colonne en numérique
    294. 

  294.                 $col = array_search(strtolower($cols), $l);
    295. 

  295.                 $tab[] = $cellules[$row][$col];
    296. 

  296.             }
    297. 

  297.             // S'il n'y a qu'un élément, on additionne son contenu
    298. 

  298.             if (count($tab) == 1) 
    299. 

  299.                 $somme += $tab[0]->getContent() * 1;    
    300. 

  300.             else 
    301. 

  301.                 $cells = $this->parent->getRangeCells($tab[0], $tab[1], '_setSum', '', $somme);    
    302. 

  302.         }
    303. 

  303.         // On set les propriétés
    304. 

  304.         $this->formula = $f;
    305. 

  305.         $this->contenu = $somme;
    306. 

  306.         $this->type = 'float';
    307. 

  307.         return $this->type;
    308. 

  308.     }
