  222.     public function save($in_file = true) {
    223. 

  223.         $this->_saveFile();
    224. 

  224.         $zip = new EasyZIP();
    225. 

  225.         // On ajoute tous les fichiers au ZIP
    226. 

  226.         if ($handle = opendir($this->pathSave)) { 
    227. 

  227.             while (false !== ($filename = readdir($handle))) {
    228. 

  228.                 if ($filename != '.' && $filename != '..'){
    229. 

  229.                     if (is_dir($this->pathSave.'/'.$filename))
    230. 

  230.                         $zip->addDir($this->pathSave, $filename);
    231. 

  231.                     else
    232. 

  232.                         $zip->addFile($filename, $this->pathSave.'/');
    233. 

  233.                 }
    234. 

  234.             }
    235. 

  235.             closedir($handle);
    236. 

  236.         }
    237. 

  237.         $fileName = ($in_file) ? $this->documentName : '';
    238. 

  238.         $result = $zip->zipFile($fileName);
    239. 

  239.         // On supprime le répertoire généré
    240. 

  240.         if (!$this->keepGeneratedRep)
    241. 

  241.             Fonction::delDir($this->pathSave);
    242. 

  242.         // On retourne le fichier ZIP
    243. 

  243.         return $result;
    244. 

  244.     }
