  147.     protected function _beforeSave() {

  148.         $styles = array(

  149.             'col' => array(), 'colused' => array(),

  150.             'row' => array(), 'rowused' => array(),

  151.             'cel' => array(), 'celused' => array(),
    152. 

  152.             'gra' => array(), 'graused' => array(),
    153. 

  153.             'leg' => array(), 'legused' => array(),
    154. 

  154.             'tab' => array(), 'tabused' => array(),
    155. 

  155.             'num' => array(), 'numused' => array()
    156. 

  156.         );
    157. 

  157.         $sheetsCols = array();
    158. 

  158.         $fonts = array();
    159. 

  159.         $maxCol = 0;
    160. 

  160.         // $colDef = co + "le chiffre UN" (et non pas "la lettre L")
    161. 

  161.         $colDef = 'co1';

  162.         // 1er parsage, récupération des styles, nb de lignes max et nb de colonnes max
    163. 

  163.         foreach ($this->sheets as $key => $sheet){
    164. 

  164.             $cells = $sheet->getCells();

  166.             // Gestion des styles des cellules

  167.             foreach ($cells as $row => $obj) {

  168.                 foreach ($obj as $col => $cell) {
    169. 

  169.                     $used = $this->_getUsedCellStyles($cell, $styles);
    170. 

  170.                     $cell->setStyleName($used);

  171.                     $fonts[] = $cell->getFontFamily();

  172.                     if (!isset($cols[$col]) || $cols[$col] == '' || ($used['col'] != $colDef))
    173. 

  173.                         $cols[$col] = $used['col'];
    174. 

  174.                 }
    175. 

  175.             }
    176. 

  176.             if ($maxCol < $sheet->getNbColsMax()) {
    177. 

  177.                 $maxCol = $sheet->getNbColsMax();
    178. 

  178.             }
    179. 

  179.             $sheetsCols[$key] = Fonction::array_count_followed_values($cols, $colDef, $maxCol);
    180. 

  180.             // Gestion des styles des images et graphiques
    181. 

  181.             foreach ($sheet->getPictures() as $pict) {
    182. 

  182.                 $used = $this->_getUsedGraphStyles($pict, $styles);
    183. 

  183.                 $pict->setStyle = $used['gra'];

  184.                 $pict->setTextStyle = $used['text'];
    185. 

  185.             }
    186. 

  186.         }
    187. 

  187.         $this->_insertFontFamilies($fonts);
    188. 

  188.         $this->_insertStyles($styles);
    189. 

  189.         // 2e parsage, insertion des cellules
    190. 

  190.         foreach ($this->sheets as $key => $sheet){
    191. 

  191.             $cells = $sheet->getCells();
    192. 

  192.             foreach ($sheetsCols[$key] as $col) {
    193. 

  193.                 $tbCol = $this->_addTableElement('table-column', null, $sheet->getXML());
    194. 

  194.                 $tbCol->setAttribute('table:style-name', $col['value']);
    195. 

  195.                 if ($col['nb'] > 1)
    196. 

  196.                     $tbCol->setAttribute('table:number-columns-repeated', $col['nb']);
    197. 

  197.             }
    198. 

  198.             $rowCourant = 0;
    199. 

  199.             ksort($cells);
    200. 

  200.             // Insertion des cellules
    201. 

  201.             foreach ($cells as $row => $obj) {
    202. 

  202.                 ksort($obj);
    203. 

  203.                 $cell = current($obj);
    204. 

  204.                 $st = $cell->getStyleName();
    205. 

  205.                 if ($rowCourant < $row - 1) {
    206. 

  206.                     $tbRow = $this->_addTableElement('table-row', null, $sheet->getXML());
    207. 

  207.                     $tbRow->setAttribute('table:number-rows-repeated', ($row - $rowCourant - 1));
    208. 

  208.                     $tbCell = $this->_addTableElement('table-cell', null, $tbRow);
    209. 

  209.                     $tbCell->setAttribute('table:style-name', 'ce1');
    210. 

  210.                 }
    211. 

  211.                 $tbRow = $this->_addTableElement('table-row', null, $sheet->getXML());
    212. 

  212.                 $tbRow->setAttribute('table:style-name', $st['row']);
    213. 

  213.                 $colCourant = 0;
    214. 

  214.                 foreach ($obj as $col => $cell) {
    215. 

  215.                     $st = $cell->getStyleName();
    216. 

  216.                     // On regarde s'il y a des cellules vides entre deux
    217. 

  217.                     if ($colCourant < $col - 1) {
    218. 

  218.                         $tbCell = $this->_addTableElement('table-cell', null, $tbRow);
    219. 

  219.                         $tbCell->setAttribute('table:number-columns-repeated', ($col - $colCourant - 1));
    220. 

  220.                     }
    221. 

  221.                     // On ajoute la cellule de contenu
    222. 

  222.                     $tbCell = $this->_addTableElement('table-cell', null, $tbRow);
    223. 

  223.                     $tbCell->setAttribute('table:style-name', $st['cel']);
    224. 

  224.                     if ($cell->getFormula() != '') {
    225. 

  225.                         $tbCell->setAttribute('table:formula', 'oooc:'.$cell->getFormula());
    226. 

  226.                     }
    227. 

  227.                     if ($cell->getSpannedRows() > 1) {
    228. 

  228.                         $tbCell->setAttribute('table:number-rows-spanned', $cell->getSpannedRows());
    229. 

  229.                     }
    230. 

  230.                     if ($cell->getSpannedCols() > 1) {
    231. 

  231.                         $tbCell->setAttribute('table:number-columns-spanned', $cell->getSpannedCols());
    232. 

  232.                     }
    233. 

  233.                     if ($cell->getType() == 'float') {
    234. 

  234.                         $tbCell->setAttribute('office:value-type', $cell->getType());
    235. 

  235.                         $tbCell->setAttribute('office:value', $cell->getContent());
    236. 

  236.                     }
    237. 

  237.                     if ($cell->getContent() != '') {
    238. 

  238.                         $content = $this->_addTextElement('p', $cell->getContent(), $tbCell);
    239. 

  239.                     }
    240. 

  240.                     $colCourant = $col;
    241. 

  241.                 }
    242. 

  242.                 $rowCourant = $row;
    243. 

  243.             }
    244. 

  244.             // Insertion des graphiques et images
    245. 

  245.             foreach ($sheet->getPictures() as $pict) {
    246. 

  246.                 copy($pict->getPathName(), $this->pathSave.'/Pictures/'.$pict->getName());
    247. 

  247.                 $this->parent->manifest->addFileEntry($pict->getType(), 'Pictures/'.$pict->getName());
    248. 

  248.                 // Frame
    249. 

  249.                 $tbFrame = $this->_addDrawElement('frame', null, $tbCell);
    250. 

  250.                 $tbFrame->setAttribute('table:end-cell-address', Fonction::checkAttribute($pict->getCellAddress()));
    251. 

  251.                 $tbFrame->setAttribute('table:end-x', $pict->getXEnd());
    252. 

  252.                 $tbFrame->setAttribute('table:end-y', $pict->getYEnd());
    253. 

  253.                 $tbFrame->setAttribute('draw:z-index', $pict->getZIndex());
    254. 

  254.                 $tbFrame->setAttribute('draw:name', Fonction::checkAttribute($pict->getName()));
    255. 

  255.                 $tbFrame->setAttribute('draw:style-name', $pict->getStyle());
    256. 

  256.                 $tbFrame->setAttribute('draw:text-style-name', $pict->getTextStyle());
    257. 

  257.                 $tbFrame->setAttribute('svg:width', $pict->getWidth());
    258. 

  258.                 $tbFrame->setAttribute('svg:height', $pict->getHeight());
    259. 

  259.                 $tbFrame->setAttribute('svg:x', $pict->getSVGX());
    260. 

  260.                 $tbFrame->setAttribute('svg:y', $pict->getSVGY());
    261. 

  261.                 // Image
    262. 

  262.                 $tbImage = $this->_addDrawElement('image', null, $tbFrame);
    263. 

  263.                 $tbImage->setAttribute('xlink:href', Fonction::checkAttribute($pict->getPath()));
    264. 

  264.                 $tbImage->setAttribute('xlink:type', $pict->getType());
    265. 

  265.                 $tbImage->setAttribute('xlink:show', $pict->getShow());
    266. 

  266.                 $tbImage->setAttribute('xlink:actuate', $pict->getActuate());
    267. 

  267.             }    
    268. 

  268.         }
    269. 

  269.         return true;        
    270. 

  270.     }
