  146.     public function __construct($core = '', $xpath = '', $picture_path = '', $sheet_name = '') {
    147. 

  147.         if ($core && $xpath) {
    148. 

  148.             $this->load('picture', $core, $xpath);
    149. 

  149.             $this->root = $xpath->query('//office:spreadsheet')->item(0);
    150. 

  150.             $path = explode('/', Fonction::removeLastSlash($picture_path));
    151. 

  151.             $name = array_pop($path);
    152. 

  152.             //$path = (count($path) == 0) ? $path = '' : implode('/', $path);
    153. 

  153.             $this->picturePath = implode('/', $path);
    154. 

  154.             $this->name = $name;
    155. 

  155.             $this->sheetAddress = $sheet_name;
    156. 

  156.             $this->type = 'simple';
    157. 

  157.             $this->show = 'embed';
    158. 

  158.             $this->actuate = 'onLoad';
    159. 

  159.             $this->type = 'image/gif';
    160.

  161.         }
